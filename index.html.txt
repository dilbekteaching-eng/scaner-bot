        body { margin: 0; padding: 0; background-color: #1a1a1a; color: white; font-family: sans-serif; display: flex; flex-direction: column; align-items: center; justify-content: center; height: 100vh; overflow: hidden; }

    <div id="loading" class="loading">Kamera yuklanmoqda...</div>
    <video id="video"></video>
    <div id="error" class="error-log"></div>
    <button id="retryBtn" class="btn-retry" onclick="location.reload()">Qayta yuklash</button>

    <script>
        const tg = window.Telegram.WebApp;
        const video = document.getElementById('video');
        const loading = document.getElementById('loading');
        const errorEl = document.getElementById('error');
        const retryBtn = document.getElementById('retryBtn');

        tg.expand();

        async function startScanner() {
            try {
                const codeReader = new ZXing.BrowserMultiFormatReader();
                const videoInputDevices = await ZXing.BrowserCodeReader.listVideoInputDevices();
                
                if (videoInputDevices.length === 0) {
                    throw new Error("Kamera topilmadi!");
                }

                // Orqa kamerani tanlashga harakat qilamiz
                const selectedDeviceId = videoInputDevices.length > 1 ? videoInputDevices[1].deviceId : videoInputDevices[0].deviceId;

                loading.innerText = "Kamera tayyor. Shtrix-kodni ko'rsating.";
                video.style.display = "block";

                codeReader.decodeFromVideoDevice(selectedDeviceId, 'video', (result, err) => {
                    if (result) {
                        console.log(result.text);
                        tg.sendData(result.text);
                        tg.close();
                    }
                    if (err && !(err instanceof ZXing.NotFoundException)) {
                        console.error(err);
                    }
                });

            } catch (e) {
                loading.style.display = "none";
                errorEl.innerText = "Xatolik: " + e.message;
                retryBtn.style.display = "block";
            }
        }

        // Kutubxona yuklanishini kutamiz
        window.onload = () => {
            if (typeof ZXing !== 'undefined') {
                startScanner();
            } else {
                errorEl.innerText = "Skaner kutubxonasi yuklanmadi. Internetni tekshiring.";
                retryBtn.style.display = "block";
            }
        };
    </script>
