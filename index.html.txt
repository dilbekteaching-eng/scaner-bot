    <div id="log">Kamera yuklanishini kuting...</div>

    <script>
        const tg = window.Telegram.WebApp;
        tg.expand();
        const logDiv = document.getElementById('log');

        async function initScanner() {
            try {
                const codeReader = new ZXing.BrowserMultiFormatReader();
                logDiv.innerText = "Kamera ruxsat so'ramoqda...";

                codeReader.decodeFromVideoDevice(null, 'video', (result, err) => {
                    if (result) {
                        tg.sendData(result.text);
                        tg.close();
                    }
                });
                logDiv.innerText = "Shtrix-kodni ko'rsating";
            } catch (e) {
                logDiv.innerText = "XATO: " + e.message;
            }
        }

        // Kutubxona yuklanganini tekshirish
        window.onload = () => {
            if (typeof ZXing !== 'undefined') {
                initScanner();
            } else {
                logDiv.innerText = "Kutubxona yuklanmadi! Internetni tekshiring.";
            }
        };
    </script>
